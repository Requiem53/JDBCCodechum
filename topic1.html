MySQL CRUD

CRUD stands for Create, Read, Update, and Delete. It is a set of basic operations that are essential for working with any database system. These operations allow you to interact with the data stored in a database.

  • Create: Insert new data into the database.
  • Read: Retrieve and fetch data from the database.
  • Update: Modify existing data in the database.
  • Delete: Remove data from the database.

CRUD operations are the building blocks of many applications, enabling developers to interact with the database efficiently. In this guide, we will focus on implementing these operations in Java using MySQL as the database system.

Connecting to MySQL Database

Before performing CRUD operations, you need to establish a connection to the MySQL database. To do this, you'll need the MySQL Connector/J library, which provides a JDBC driver for connecting to MySQL databases.

Here's how you can connect to a MySQL database in Java:

1	import java.sql.Connection;
2	import java.sql.DriverManager;
3	import java.sql.SQLException;
4	
5	public class MySQLConnection {
6	    private static final String URL = "jdbc:mysql://localhost:3306/your_database_name";
7	    private static final String USER = "your_username";
8	    private static final String PASSWORD = "your_password";
9	
10	    public static Connection getConnection() {
11	        Connection connection = null;
12	        try {
13	            Class.forName("com.mysql.cj.jdbc.Driver");
14	            connection = DriverManager.getConnection(URL, USER, PASSWORD);
15	            System.out.println("Connected to the database!");
16	        } catch (ClassNotFoundException | SQLException e) {
17	            e.printStackTrace();
18	        }
19	        return connection;
20	    }
21	}

Replace your_database_name, your_username, and your_password with your actual database details.

Creating Database and Tables

Let's create a simple table named users with columns id, name, and email.

1	import java.sql.Connection;
2	import java.sql.DriverManager;
3	import java.sql.SQLException;
4	import java.sql.Statement;
5	
6	public class CreateTable {
7	    public static void main(String[] args) {
8	        try (Connection connection = MySQLConnection.getConnection();
9	             Statement statement = connection.createStatement()) {
10	            String createTableQuery = "CREATE TABLE IF NOT EXISTS users ("
11	                                    + "id INT AUTO_INCREMENT PRIMARY KEY,"
12	                                    + "name VARCHAR(50) NOT NULL,"
13	                                    + "email VARCHAR(100) NOT NULL)";
14	            statement.execute(createTableQuery);
15	            System.out.println("Table created successfully!");
16	        } catch (SQLException e) {
17	            e.printStackTrace();
18	        }
19	    }
20	}

The above code will create a table named users if it doesn't already exist.

Inserting Data into Tables

Next, let's insert data into the users table.

1	import java.sql.Connection;
2	import java.sql.DriverManager;
3	import java.sql.PreparedStatement;
4	import java.sql.SQLException;
5	
6	public class InsertData {
7	    public static void main(String[] args) {
8	        try (Connection connection = MySQLConnection.getConnection();
9	             PreparedStatement preparedStatement = connection.prepareStatement(
10	                     "INSERT INTO users (name, email) VALUES (?, ?)")) {
11	            // Set values for name and email
12	            String name = "John Doe";
13	            String email = "john@example.com";
14	
15	            preparedStatement.setString(1, name);
16	            preparedStatement.setString(2, email);
17	
18	            int rowsInserted = preparedStatement.executeUpdate();
19	            if (rowsInserted > 0) {
20	                System.out.println("Data inserted successfully!");
21	            }
22	        } catch (SQLException e) {
23	            e.printStackTrace();
24	        }
25	    }
26	}

The code above will insert a new record into the users table.

Retrieving Data from Tables

To retrieve data from the users table, you can use a simple SELECT query.

1	import java.sql.Connection;
2	import java.sql.DriverManager;
3	import java.sql.ResultSet;
4	import java.sql.SQLException;
5	import java.sql.Statement;
6	
7	public class RetrieveData {
8	    public static void main(String[] args) {
9	        try (Connection connection = MySQLConnection.getConnection();
10	             Statement statement = connection.createStatement()) {
11	
12	            String selectQuery = "SELECT * FROM users";
13	            ResultSet resultSet = statement.executeQuery(selectQuery);
14	
15	            while (resultSet.next()) {
16	                int id = resultSet.getInt("id");
17	                String name = resultSet.getString("name");
18	                String email = resultSet.getString("email");
19	
20	                System.out.println("ID: " + id + ", Name: " + name + ", Email: " + email);
21	            }
22	        } catch (SQLException e) {
23	            e.printStackTrace();
24	        }
25	    }
26	}

The above code will retrieve all data from the users table and print it.

Updating Data in Tables

To update existing data in the users table, you can use the UPDATE statement.

1	import java.sql.Connection;
2	import java.sql.DriverManager;
3	import java.sql.PreparedStatement;
4	import java.sql.SQLException;
5	
6	public class UpdateData {
7	    public static void main(String[] args) {
8	        try (Connection connection = MySQLConnection.getConnection();
9	             PreparedStatement preparedStatement = connection.prepareStatement(
10	                     "UPDATE users SET name = ?, email = ? WHERE id = ?")) {
11	
12	            // Set new values for name, email, and the user ID to update
13	            String newName = "Jane Doe";
14	            String newEmail = "jane@example.com";
15	            int userIdToUpdate = 1;
16	
17	            preparedStatement.setString(1, newName);
18	            preparedStatement.setString(2, newEmail);
19	            preparedStatement.setInt(3, userIdToUpdate);
20	
21	            int rowsUpdated = preparedStatement.executeUpdate();
22	            if (rowsUpdated > 0) {
23	                System.out.println("Data updated successfully!");
24	            }
25	        } catch (SQLException e) {
26	            e.printStackTrace();
27	        }
28	    }
29	}

The code above will update the name and email for the user with ID 1 in the users table.

Deleting Data from Tables

To delete data from the users table, you can use the DELETE statement.

1	import java.sql.Connection;
2	import java.sql.DriverManager;
3	import java.sql.PreparedStatement;
4	import java.sql.SQLException;
5	
6	public class DeleteData {
7	    public static void main(String[] args) {
8	        try (Connection connection = MySQLConnection.getConnection();
9	             PreparedStatement preparedStatement = connection.prepareStatement(
10	                     "DELETE FROM users WHERE id = ?")) {
11	
12	            // Set the user ID to delete
13	            int userIdToDelete = 1;
14	
15	            preparedStatement.setInt(1, userIdToDelete);
16	
17	            int rowsDeleted = preparedStatement.executeUpdate();
18	            if (rowsDeleted > 0) {
19	                System.out.println("Data deleted successfully!");
20	            }
21	        } catch (SQLException e) {
22	            e.printStackTrace();
23	        }
24	    }
25	}

The code above will delete the user with ID 1 from the users table.
